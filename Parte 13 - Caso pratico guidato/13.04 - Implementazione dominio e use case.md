L’obiettivo non è ancora costruire tutta l’API o tutta la UI.

# PercentualeSconto

`App.Domain/Documenti/PercentualeSconto.cs`:

```csharp id="mk91sc"
namespace App.Domain.Documenti;

public readonly record struct PercentualeSconto
{
    public PercentualeSconto(decimal value)
    {
        if (value < 0 || value > 100)
        {
            throw new ArgumentOutOfRangeException(
                nameof(value),
                "Lo sconto deve essere compreso tra 0 e 100.");
        }

        Value = value;
    }

    public decimal Value { get; }

    public decimal FattoreMoltiplicativo =>
        1 - Value / 100;
}
```


```csharp id="l2jami"
namespace App.Domain.Common;

public readonly record struct Periodo
{
    public Periodo(DateOnly dataDa, DateOnly dataA)
    {
        if (dataDa > dataA)
        {
            throw new ArgumentException(
                "La data iniziale non può essere successiva alla data finale.");
        }

        DataDa = dataDa;
        DataA = dataA;
    }

    public DateOnly DataDa { get; }

    public DateOnly DataA { get; }

    public bool Contiene(DateOnly data)
    {
        return data >= DataDa && data <= DataA;
    }
}
```

```csharp id="d56jfb"
namespace App.Domain.Catalogo;

public class Libro : Entity
{
    private readonly List<LibroAutore> autori = [];

    private Libro()
    {
    }

    public Libro(
        string titolo,
        string? isbn,
        decimal prezzo,
        DateOnly? dataPubblicazione,
        int? collanaId)
    {
        ImpostaDati(
            titolo,
            isbn,
            prezzo,
            dataPubblicazione,
            collanaId);

        Attivo = true;
    }

    public string Titolo { get; private set; } = "";

    public string? Isbn { get; private set; }

    public decimal Prezzo { get; private set; }

    public DateOnly? DataPubblicazione { get; private set; }

    public bool Attivo { get; private set; }

    public int? CollanaId { get; private set; }

    public Collana? Collana { get; private set; }

    public IReadOnlyCollection<LibroAutore> Autori =>
        autori.AsReadOnly();

    public void ImpostaDati(
        string titolo,
        string? isbn,
        decimal prezzo,
        DateOnly? dataPubblicazione,
        int? collanaId)
    {
        if (string.IsNullOrWhiteSpace(titolo))
        {
            throw new DomainException("Il titolo del libro è obbligatorio.");
        }

        if (prezzo < 0)
        {
            throw new DomainException("Il prezzo del libro non può essere negativo.");
        }

        Titolo = titolo.Trim();
        Isbn = string.IsNullOrWhiteSpace(isbn)
            ? null
            : isbn.Trim();

        Prezzo = prezzo;
        DataPubblicazione = dataPubblicazione;
        CollanaId = collanaId;
    }

    public void Attiva()
    {
        Attivo = true;
    }

    public void Disattiva()
    {
        Attivo = false;
    }
}
```

## DocumentoRiga

`App.Domain/Documenti/DocumentoRiga.cs`:

```csharp id="iy7szm"
using App.Domain.Common;

namespace App.Domain.Documenti;

public class DocumentoRiga : Entity
{
    private DocumentoRiga()
    {
    }

    public DocumentoRiga(
        int libroId,
        string descrizione,
        int quantita,
        decimal prezzoUnitario,
        decimal scontoPercentuale)
    {
        if (libroId <= 0)
        {
            throw new DomainException("Il libro della riga è obbligatorio.");
        }

        if (string.IsNullOrWhiteSpace(descrizione))
        {
            throw new DomainException("La descrizione della riga è obbligatoria.");
        }

        if (quantita <= 0)
        {
            throw new DomainException("La quantità deve essere maggiore di zero.");
        }

        if (prezzoUnitario < 0)
        {
            throw new DomainException("Il prezzo unitario non può essere negativo.");
        }

        PercentualeSconto sconto = new(scontoPercentuale);

        LibroId = libroId;
        Descrizione = descrizione.Trim();
        Quantita = quantita;
        PrezzoUnitario = prezzoUnitario;
        ScontoPercentuale = sconto.Value;
        TotaleLordo = quantita * prezzoUnitario;
        TotaleRiga = TotaleLordo * sconto.FattoreMoltiplicativo;
    }

    public int DocumentoId { get; private set; }

    public Documento Documento { get; private set; } = null!;

    public int LibroId { get; private set; }

    public string Descrizione { get; private set; } = "";

    public int Quantita { get; private set; }

    public decimal PrezzoUnitario { get; private set; }

    public decimal ScontoPercentuale { get; private set; }

    public decimal TotaleLordo { get; private set; }

    public decimal TotaleRiga { get; private set; }
}
```

La riga calcola il proprio totale.

Questo evita errori come:

## Documento

`App.Domain/Documenti/Documento.cs`:

```csharp id="lkd7z8"
using App.Domain.Common;

namespace App.Domain.Documenti;

public class Documento : Entity
{
    private readonly List<DocumentoRiga> righe = [];

    private Documento()
    {
    }

    public Documento(
        string numero,
        DateOnly data,
        int clienteId,
        TipoDocumento tipoDocumento)
    {
        if (string.IsNullOrWhiteSpace(numero))
        {
            throw new DomainException("Il numero documento è obbligatorio.");
        }

        if (clienteId <= 0)
        {
            throw new DomainException("Il cliente è obbligatorio.");
        }

        Numero = numero.Trim();
        Data = data;
        ClienteId = clienteId;
        TipoDocumento = tipoDocumento;
        Stato = StatoDocumento.Bozza;
    }

    public string Numero { get; private set; } = "";

    public DateOnly Data { get; private set; }

    public int ClienteId { get; private set; }

    public TipoDocumento TipoDocumento { get; private set; }

    public StatoDocumento Stato { get; private set; }

    public IReadOnlyCollection<DocumentoRiga> Righe =>
        righe.AsReadOnly();

    public decimal TotaleImponibile { get; private set; }

    public decimal TotaleSconto { get; private set; }

    public decimal TotaleDocumento { get; private set; }

    public void AggiungiRiga(
        int libroId,
        string descrizione,
        int quantita,
        decimal prezzoUnitario,
        decimal scontoPercentuale)
    {
        VerificaModificabile();

        DocumentoRiga riga = new(
            libroId,
            descrizione,
            quantita,
            prezzoUnitario,
            scontoPercentuale);

        righe.Add(riga);

        RicalcolaTotali();
    }

    public void Conferma()
    {
        if (Stato != StatoDocumento.Bozza)
        {
            throw new DomainException(
                "Solo un documento in bozza può essere confermato.");
        }

        if (righe.Count == 0)
        {
            throw new DomainException(
                "Non è possibile confermare un documento senza righe.");
        }

        Stato = StatoDocumento.Confermato;
    }

    public void Annulla()
    {
        if (Stato == StatoDocumento.Annullato)
        {
            return;
        }

        Stato = StatoDocumento.Annullato;
    }

    private void VerificaModificabile()
    {
        if (Stato != StatoDocumento.Bozza)
        {
            throw new DomainException(
                "È possibile modificare solo documenti in bozza.");
        }
    }

    private void RicalcolaTotali()
    {
        decimal totaleLordo = righe.Sum(x => x.TotaleLordo);
        decimal totaleNetto = righe.Sum(x => x.TotaleRiga);

        TotaleImponibile = totaleLordo;
        TotaleDocumento = totaleNetto;
        TotaleSconto = totaleLordo - totaleNetto;
    }
}
```

Questa entity protegge alcune regole fondamentali.

```text id="kh1ko0"
Il documento nasce in bozza.
Solo un documento in bozza può essere modificato.
Un documento senza righe non può essere confermato.
I totali sono calcolati dal dominio.
```

# Repository interface

Nella Clean Architecture classica, l’Application Layer definisce interfacce di repository e l’Infrastructure Layer le implementa.

Nel nostro progetto possiamo usare due approcci.

## Approccio A: `IAppDbContext`

È più semplice.

```text id="skt8cr"
Application usa IAppDbContext.
Infrastructure implementa AppDbContext con EF Core.
```

Vantaggi:

```text id="q71n94"
meno classi
meno astrazione inutile
query LINQ dirette
didatticamente semplice
```

Svantaggi:

```text id="nehl7m"
Application conosce concetti EF Core come DbSet
Clean Architecture meno pura
```

## Approccio B: repository specifici

È più esplicito.

```text id="ga2plp"
ILibroRepository
IClienteRepository
IDocumentoRepository
```

Vantaggi:

```text id="hi4fmr"
Application non conosce EF Core
dipendenze più pulite
test più facili con fake repository
```

Svantaggi:

```text id="gsexq1"
più codice
rischio repository banali
possibile duplicazione di query
```

Per questo capitolo mostriamo il repository specifico almeno per i documenti, perché è didatticamente utile.

# Implementazione repository

`App.Infrastructure/Documenti/DocumentoRepository.cs`:

```csharp id="y9efl2"
using App.Application.Documenti;
using App.Domain.Documenti;
using App.Infrastructure.Data;
using Microsoft.EntityFrameworkCore;

namespace App.Infrastructure.Documenti;

public class DocumentoRepository : IDocumentoRepository
{
    private readonly AppDbContext dbContext;

    public DocumentoRepository(AppDbContext dbContext)
    {
        this.dbContext = dbContext;
    }

    public Task<Documento?> GetByIdAsync(
        int id,
        CancellationToken cancellationToken)
    {
        return dbContext.Documenti
            .Include(x => x.Righe)
            .FirstOrDefaultAsync(x => x.Id == id, cancellationToken);
    }

    public async Task AddAsync(
        Documento documento,
        CancellationToken cancellationToken)
    {
        await dbContext.Documenti.AddAsync(
            documento,
            cancellationToken);
    }

    public Task<bool> ClienteExistsAsync(
        int clienteId,
        CancellationToken cancellationToken)
    {
        return dbContext.Clienti
            .AnyAsync(x => x.Id == clienteId, cancellationToken);
    }

    public Task<bool> LibroExistsAsync(
        int libroId,
        CancellationToken cancellationToken)
    {
        return dbContext.Libri
            .AnyAsync(x => x.Id == libroId && x.Attivo, cancellationToken);
    }

    public async Task<string> GetDescrizioneLibroAsync(
        int libroId,
        CancellationToken cancellationToken)
    {
        string? titolo = await dbContext.Libri
            .Where(x => x.Id == libroId)
            .Select(x => x.Titolo)
            .FirstOrDefaultAsync(cancellationToken);

        return titolo ?? "";
    }

    public async Task<decimal> GetPrezzoLibroAsync(
        int libroId,
        CancellationToken cancellationToken)
    {
        decimal? prezzo = await dbContext.Libri
            .Where(x => x.Id == libroId)
            .Select(x => (decimal?)x.Prezzo)
            .FirstOrDefaultAsync(cancellationToken);

        return prezzo ?? 0m;
    }

    public async Task SaveChangesAsync(
        CancellationToken cancellationToken)
    {
        await dbContext.SaveChangesAsync(cancellationToken);
    }
}
```

```csharp id="vxo82t"
public class Result
{
    protected Result(
        bool success,
```

```csharp id="gg1fdz"
using App.Application.Common;

namespace App.Application.Documenti;

public static class CreateDocumentoValidator
{
    public static Result Validate(CreateDocumentoCommand command)
    {
        if (command.ClienteId <= 0)
        {
            return Result.Fail("Il cliente è obbligatorio.");
        }

        if (command.Righe.Count == 0)
        {
            return Result.Fail(
                "Il documento deve contenere almeno una riga.");
        }

        foreach (CreateDocumentoRigaCommand riga in command.Righe)
        {
            if (riga.LibroId <= 0)
            {
                return Result.Fail("Ogni riga deve avere un libro valido.");
            }

            if (riga.Quantita <= 0)
            {
                return Result.Fail(
                    "La quantità deve essere maggiore di zero.");
            }

            if (riga.PrezzoUnitario is < 0)
            {
                return Result.Fail(
                    "Il prezzo unitario non può essere negativo.");
            }

            if (riga.ScontoPercentuale < 0 ||
                riga.ScontoPercentuale > 100)
            {
                return Result.Fail(
                    "Lo sconto deve essere compreso tra 0 e 100.");
            }
        }

        return Result.Ok();
    }
}
```

# CreateDocumentoHandler

`App.Application/Documenti/CreateDocumentoHandler.cs`:

```csharp id="d125hg"
using App.Application.Common;
using App.Domain.Common;
using App.Domain.Documenti;
using Microsoft.Extensions.Logging;

namespace App.Application.Documenti;

public class CreateDocumentoHandler
{
    private readonly IDocumentoRepository documentoRepository;
    private readonly ILogger<CreateDocumentoHandler> logger;

    public CreateDocumentoHandler(
        IDocumentoRepository documentoRepository,
        ILogger<CreateDocumentoHandler> logger)
    {
        this.documentoRepository = documentoRepository;
        this.logger = logger;
    }

    public async Task<Result<int>> HandleAsync(
        CreateDocumentoCommand command,
        CancellationToken cancellationToken)
    {
        Result validation = CreateDocumentoValidator.Validate(command);

        if (!validation.Success)
        {
            return Result<int>.Fail(
                validation.Error!,
                validation.ErrorType);
        }

        bool clienteExists =
            await documentoRepository.ClienteExistsAsync(
                command.ClienteId,
                cancellationToken);

        if (!clienteExists)
        {
            return Result<int>.Fail(
                "Cliente non trovato.",
                ErrorType.NotFound);
        }

        string numero = await GeneraNumeroDocumentoAsync(
            command.Data,
            cancellationToken);

        Documento documento = new(
            numero,
            command.Data,
            command.ClienteId,
            command.TipoDocumento);

        foreach (CreateDocumentoRigaCommand riga in command.Righe)
        {
            bool libroExists =
                await documentoRepository.LibroExistsAsync(
                    riga.LibroId,
                    cancellationToken);

            if (!libroExists)
            {
                return Result<int>.Fail(
                    $"Libro {riga.LibroId} non trovato o non attivo.",
                    ErrorType.NotFound);
            }

            string descrizione =
                await documentoRepository.GetDescrizioneLibroAsync(
                    riga.LibroId,
                    cancellationToken);

            decimal prezzo =
                riga.PrezzoUnitario
                ?? await documentoRepository.GetPrezzoLibroAsync(
                    riga.LibroId,
                    cancellationToken);

            documento.AggiungiRiga(
                riga.LibroId,
                descrizione,
                riga.Quantita,
                prezzo,
                riga.ScontoPercentuale);
        }

        try
        {
            await documentoRepository.AddAsync(
                documento,
                cancellationToken);

            await documentoRepository.SaveChangesAsync(
                cancellationToken);

            logger.LogInformation(
                "Documento creato con Id {DocumentoId} e numero {Numero}",
                documento.Id,
                documento.Numero);

            return Result<int>.Ok(documento.Id);
        }
        catch (DomainException ex)
        {
            return Result<int>.Fail(
                ex.Message,
                ErrorType.Business);
        }
    }

    private Task<string> GeneraNumeroDocumentoAsync(
        DateOnly data,
        CancellationToken cancellationToken)
    {
        // Versione didattica.
        // In un progetto reale servirebbe una numerazione robusta,
        // transazionale e concorrente.
        string numero = $"DOC-{data:yyyyMMdd}-{Guid.NewGuid():N}"[..22];

        return Task.FromResult(numero);
    }
}
```

Nota didattica importante:

```text id="gmvq6q"
La generazione del numero documento qui è volutamente semplice.
In un gestionale reale la numerazione richiede regole più rigorose,
transazioni e gestione della concorrenza.
```

# ConfermaDocumentoHandler

```csharp id="o0ovqx"
using App.Application.Common;
using App.Domain.Common;
using Microsoft.Extensions.Logging;

namespace App.Application.Documenti;

public class ConfermaDocumentoHandler
{
    private readonly IDocumentoRepository documentoRepository;
    private readonly ILogger<ConfermaDocumentoHandler> logger;

    public ConfermaDocumentoHandler(
        IDocumentoRepository documentoRepository,
        ILogger<ConfermaDocumentoHandler> logger)
    {
        this.documentoRepository = documentoRepository;
        this.logger = logger;
    }

    public async Task<Result> HandleAsync(

                cancellationToken);

                "Documento {DocumentoId} confermato",
                documento.Id);

            return Result.Ok();
        }
        catch (DomainException ex)
        {
            return Result.Fail(
                ex.Message,
                ErrorType.Business);
        }
    }
```

```csharp id="uz1erz"
using App.Application.Documenti;
using App.Infrastructure.Data;
using Microsoft.EntityFrameworkCore;

namespace App.Infrastructure.Documenti;

public class DocumentoQueries : IDocumentoQueries
{
    private readonly AppDbContext dbContext;

    public DocumentoQueries(AppDbContext dbContext)
    {
        this.dbContext = dbContext;
    }

    public Task<DocumentoDetailDto?> GetDocumentoByIdAsync(
        int id,
        CancellationToken cancellationToken)
    {
        return dbContext.Documenti
            .AsNoTracking()
            .Where(x => x.Id == id)
            .Select(x => new DocumentoDetailDto(
                x.Id,
                x.Numero,
                x.Data,
                x.ClienteId,
                x.TipoDocumento,
                x.Stato,
                x.TotaleImponibile,
                x.TotaleSconto,
                x.TotaleDocumento,
                x.Righe
                    .OrderBy(r => r.Id)
                    .Select(r => new DocumentoRigaDto(
                        r.Id,
                        r.LibroId,
                        r.Descrizione,
                        r.Quantita,
                        r.PrezzoUnitario,
                        r.ScontoPercentuale,
                        r.TotaleRiga))
                    .ToList()))
            .FirstOrDefaultAsync(cancellationToken);
    }
}
```

# Quando non esagerare

Non ogni operazione deve diventare una struttura enorme.

Per un CRUD semplice, può bastare:

```text id="o2vibv"
servizio applicativo
metodo CreateAsync
metodo UpdateAsync
metodo GetAsync
```

Command/query/handler diventano più utili quando:

```text id="tt4ig6"
il caso d’uso ha regole
serve testarlo bene
la validazione è articolata
la modifica coinvolge più entity
serve distinguere lettura e scrittura
```

Regola pratica:

```text id="tf7qhh"
Usa command/query/handler dove aumentano chiarezza.
Non usarli per aggiungere cerimonia inutile.
```

# Errori comuni

## Entity anemiche

Sbagliato:

```text id="pyxcij"
Documento ha solo get/set.
Tutte le regole sono sparse nei servizi.
```

Meglio:

```text id="hh1rvj"
Documento.Conferma()
Documento.AggiungiRiga()
Documento.Annulla()
```

Le regole principali devono stare vicino ai dati che proteggono.

---

## Business logic negli endpoint

Sbagliato:

```text id="g1yvbk"
MapPost contiene validazioni, query, calcoli,
creazione entity, salvataggio e logging.
```

Meglio:

```text id="tq5f5s"
endpoint → handler → dominio/repository
```

---

## Repository generico ovunque

Un `IRepository<T>` generico spesso non esprime il linguaggio del dominio.

Meglio repository specifici quando servono:

```text id="mtt01r"
IDocumentoRepository
ILibroRepository
```

---

## Usare eccezioni per errori attesi

Errore:

```text id="xbb850"
Cliente non trovato → exception
ISBN duplicato → exception
validazione input → exception
```

Meglio:

```text id="l2ocqd"
Result.NotFound
Result.Conflict
Result.Validation
```

Le eccezioni restano per casi eccezionali o violazioni del dominio non previste dal flusso normale.

---

## Query che caricano aggregate inutilmente

Per leggere una lista o un report non serve sempre caricare entity complete.

Meglio proiettare in DTO:

```csharp id="fejdtq"
.Select(x => new DocumentoListItemDto(...))
```

# Linee guida pratiche

```text id="e1nij5"
Metti le regole fondamentali nelle entity
  non sparse tra endpoint e UI.

Usa value object
  quando un valore ha regole proprie.

Usa command
  per richieste che modificano lo stato.

Usa query
  per richieste di lettura.

Usa handler
  per orchestrare il caso d’uso.

Usa repository interface
  quando vuoi isolare la persistenza nei casi di modifica.

Usa query service
  per letture ottimizzate e DTO.

Usa Result pattern
  per errori attesi.

Usa eccezioni
  per violazioni di dominio o errori imprevisti.

Scrivi test unitari
  prima sul dominio, poi sugli handler.

Non introdurre astrazioni inutili
  ogni classe deve avere un motivo chiaro.
```

# Riepilogo

In questo capitolo abbiamo iniziato l’implementazione reale del progetto finale.

Abbiamo visto:

```text id="qu71kw"
entity
value object
repository interface
command
query
handler
validazione
Result pattern
test unitari
```

La distinzione più importante è questa:

```text id="cpxjjz"
Entity
  protegge stato e regole fondamentali.

Value object
  rappresenta valori validi e coerenti.

Command
  chiede una modifica.

Query
  chiede una lettura.

Handler
  esegue un caso d’uso.

Repository
  astrae la persistenza per operazioni di modifica.

Result
  rappresenta successo o errore atteso.

Test unitari
  verificano regole e casi d’uso senza infrastruttura pesante.
```

La regola principale è questa:

```text id="inldkz"
Il dominio deve funzionare prima dell’API.
Se le regole sono corrette e testabili nel cuore dell’applicazione,
endpoint, UI, database e Docker diventano strati esterni,
non il luogo dove nasconde la logica principale.
```

Con questa base possiamo passare al capitolo successivo: implementare la persistenza con EF Core, configurare SQLite e collegare i casi d’uso al database reale.
